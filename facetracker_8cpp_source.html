265 QRect FaceTracker::findClosest(const std::vector<cv::Rect> &rects, QPoint point)
266 {
267  cv::Rect bestMatch(0,0,0,0);
268  float bestDistanceSquared = std::numeric_limits<float>::max();
269  for(std::vector<cv::Rect>::const_iterator itr = rects.begin();
270  itr != rects.end(); ++itr)
271  {
272  float dX = fabs((float)((itr->x + (itr->width/2)) - point.x()));
273  float dY = fabs((float)((itr->y + (itr->height/2)) - point.y()));
274 
275  float distSq = dX*dX + dY*dY;
276 
277  if(distSq < bestDistanceSquared)
278  {
279  bestMatch = *itr;
280  bestDistanceSquared = distSq;
281  }
282  }
283 
284  return QRect(bestMatch.x, bestMatch.y, bestMatch.width, bestMatch.height);
285 }

288 {

300

304  return;
305  }
306 
307  //To Grayscale
308  if(m_cameraFrame.channels() == 3)

311  cv::cvtColor(m_cameraFrame, cameraFrame, CV_BGRA2GRAY);
312 
313  //Histogram Equalization
314  cv::equalizeHist(cameraFrame, cameraFrame);
315 
316 #ifdef DEBUG_CAPTURE_TIMING
